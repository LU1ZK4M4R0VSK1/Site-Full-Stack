using System.Collections.Concurrent;
using System.Text;
using Microsoft.Extensions.Logging;
using System.Threading.Tasks;
using Microsoft.AspNetCore.Http;
using System.Collections.Generic;

namespace MyProject.Services
{
    public class LogService
    {
        private readonly ConcurrentDictionary<string, HttpResponse> _clients = new();
        private readonly ILogger<LogService> _logger;

        public LogService(ILogger<LogService> logger)
        {
            _logger = logger;
        }

        public void AddClient(string connectionId, HttpResponse response)
        {
            _clients.TryAdd(connectionId, response);
            _logger.LogInformation("Cliente conectado: {ConnectionId}", connectionId);
        }

        public void RemoveClient(string connectionId)
        {
            _clients.TryRemove(connectionId, out _);
            _logger.LogInformation("Cliente desconectado: {ConnectionId}", connectionId);
        }

        public async Task SendLogAsync(string log)
        {
            var disconnectedClients = new List<string>();

            foreach (var client in _clients)
            {
                try
                {
                    var message = $"data: {log}\n\n";
                    var bytes = Encoding.UTF8.GetBytes(message);
                    await client.Value.Body.WriteAsync(bytes);
                    await client.Value.Body.FlushAsync();
                }
                catch
                {
                    disconnectedClients.Add(client.Key);
                }
            }

            // Remove clientes desconectados
            foreach (var clientId in disconnectedClients)
            {
                RemoveClient(clientId);
            }
        }
    }
}
